3. Interface and cloning of object
import java.io.*;
import java.util.ArrayList;
interface printable{  
void print();  
}  
public class sample implements printable,Cloneable{  
public void print(){System.out.println("Hi dear");}  
  
public static void main(String args[]) throws
                          CloneNotSupportedException{  
sample obj1 = new sample();  
obj1.print();  
sample obj2 = (sample)obj1.clone();
obj2.print();
 }  
}
